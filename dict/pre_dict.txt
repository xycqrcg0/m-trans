# ── OCR 常见误识别修正 ──
# 长音符「ー」被 OCR 误识别
\一(?=[ぁ-んァ-ヶ])	ー	# 长音符误识别为汉字「一」
\-(?=[ぁ-んァ-ヶ])	ー	# 长音符误识别为连字符
# 句号误识别
\.\s*$	。	# 英文句号转日文句号
# 波浪号统一
~	〜	# ASCII波浪号转全角

# ── OCR 混淆字符 ──
l(?=[ぁ-んァ-ヶ])	し	# 小写L误识别为「し」

# ── 空白清理 ──
^\s+		# 去除行首空白
\s+$		# 去除行尾空白
