# 格式：源词[至少一个Tab]目标词[至少一个Tab][#注释 或 //注释]
# Format: source term[at least one Tab]target term[at least one Tab][#comment or //comment]
# 注意：源词会被当作正则表达式编译，特殊字符需转义
# 注意：line.split() 按空格分割，目标词不能含空格

# ── 标点规范化 ──
,,+	，	# 多个逗号合并
\.\.+	…	# 连续句号转省略号
!\s*!	！	# 多个感叹号合并
\?\s*\?	？	# 多个问号合并
,,	，	# 英文逗号转中文逗号

# ── 日文残留假名碎片清理 ──
[ぁ-んァ-ヶ]{1,2}$		# 清除译文末尾残留的1-2个假名
^[ぁ-んァ-ヶ]{1,2}\s		# 清除译文开头残留的1-2个假名

# ── 空白清理 ──
^\s+		# 行首空白
\s+$		# 行尾空白
